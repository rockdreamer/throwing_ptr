14 /** \brief unique_ptr that manages a single object
15  *
16  * throwing::unique_ptr is a smart pointer that owns and manages another object
17  * through a pointer and disposes of that object when the unique_ptr goes out of
18  * scope.
19  *
20  * The object is disposed of using the associated deleter when either of the
21  * following happens:
22  * - the managing unique_ptr object is destroyed
23  * - the managing unique_ptr object is assigned another pointer via operator= or
24  * reset().
25  *
26  * The object is disposed of using a potentially user-supplied deleter by
27  * calling get_deleter()(ptr). The default deleter uses the delete operator,
28  * which destroys the object and deallocates the memory.
29  *
30  * A unique_ptr may alternatively own no object, in which case it is called
31  * empty.
32  */
33 template <typename T, typename Deleter = std::default_delete<T>>
34 class unique_ptr {

36  typedef typename std::unique_ptr<T, Deleter> std_unique_ptr_type;
37  typedef typename std::unique_ptr<T, Deleter>::pointer pointer;
38  typedef typename std::unique_ptr<T, Deleter>::element_type element_type;
39  typedef typename std::unique_ptr<T, Deleter>::deleter_type deleter_type;

101 
102  /** \brief Constructs a unique_ptr by transferring ownership from u to
103  * *this, where u is constructed with a specified deleter (E).
104  *
105  * It depends upon whether E is a reference type, as following:
106  * a) if E is a reference type, this deleter is copy constructed from u's
107  * deleter (requires that this construction does not throw)
108  * b) if E is a non-reference type, this deleter is move constructed from
109  * u's deleter (requires that this construction does not throw)
110  *
111  * This constructor only participates in overload resolution if all of the
112  * following is true:
113  * a) unique_ptr<U, E>::pointer is implicitly convertible to pointer
114  * b) U is not an array type
115  * c) Either Deleter is a reference type and E is the same type as D, or
116  * Deleter is not a reference type and E is implicitly convertible to D
117  */
118  template <class U, class E>
119  unique_ptr(unique_ptr<U, E> &&u) TSP_NOEXCEPT
