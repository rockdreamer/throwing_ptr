130  "[shared_ptr][comparison]") {
131  int *ptr = new int;
132  auto t_ptr = throwing::shared_ptr<int>(ptr);
133  auto empty_t_ptr = throwing::shared_ptr<int>();
134

136  REQUIRE_FALSE(t_ptr == nullptr);

138  REQUIRE_FALSE(nullptr == t_ptr);
139

143  REQUIRE(nullptr != t_ptr);
144

148  REQUIRE((nullptr < empty_t_ptr) == std::less<int *>()(nullptr, nullptr));
149

151  REQUIRE((empty_t_ptr > nullptr) == std::less<int *>()(nullptr, nullptr));
